 42:  */
 43:   private static $error_type = array(
 44:     E_ERROR => 'Fatal error',
 45:     E_WARNING => 'Warning',
 46:     E_PARSE => 'Parse error',
 47:     E_NOTICE => 'Notice',
 48:     E_CORE_ERROR => 'Core error',
 49:     E_CORE_WARNING => 'Core warning',
 50:     E_COMPILE_ERROR => 'Compile-time error',
 51:     E_COMPILE_WARNING => 'Compile-time warning',
 52:     E_USER_ERROR => 'User-generated error',
 53:     E_USER_WARNING => 'User-generated warning',
 54:     E_USER_NOTICE => 'User-generated notice',
 55:     E_STRICT => 'Runtime notice',
 56:     E_RECOVERABLE_ERROR => 'Recoverable error'
 57:   );
 58:   private static function getErrorName($errno) {
 59:     return $error_name = (isset(self::$error_type[$errno]) ? self::$error_type[$errno] : 'Unknown error code');

 62:   /**
 63:    * @var array $php_errorlist An array holding all php errors as arrays of [$error_name,$errstr,$errfile,$errline]

 65:   public static $php_errorlist = array();

 67:   /**
 68:    * Places all php errors into the array $php_errorlist
 69:    * @param string $errno A numeric value which corresponds to the type of error (Notice, Fatal Error, User-generated warning, etc).
 70:    * @param string $errstr A string that contains the error message text, ideally including details that identify the cause of the error.
 71:    * @param string $errfile The full local path of the file which has triggered this error (such as /var/www/public_html/badscript.php).
 72:    * @param string $errline The line number where the error was generated (within the file identified by $errfile).

 78:         'name' => $error_name,

115:   public static function errorsToEmail($errno, $errstr, $errfile, $errline) {

118:     self::errorsToLog($errno, $errstr, $errfile, $errline);
119:     /* 
120:      * Stage two: find out whether we need to email a warning
121:      * to the webmaster.
122:      */

124:     $lockfile = fopen(PHP_ERROR_LOCK, 'a+');
125:     if (!$lockfile) {
126:         error_log('FAIL: fopen failed in '. __FUNCTION__ .' in '. __FILE__ .'');
127:         error_log(__FUNCTION__ .' failed! Check server logs!',1, SEND_ERROR_EMAIL_TO);
128:         echo '<p>A failure occurred, and it\'s not possible to continue.</p>';
129:         die();
130:     }
131:     $locked = flock($lockfile, LOCK_EX);
132:     if (!$locked) {
133:         error_log('FAIL: flock failed in '. __FUNCTION__ .' in '. __FILE__ .'');
134:         error_log(__FUNCTION__ .' failed! Check server logs!',1, SEND_ERROR_EMAIL_TO);
135:         echo '<p>A failure occurred, and it\'s not possible to continue.</p>';
136:         die();
137:     }
138:     rewind($lockfile);
139:     
140:     // Run through the lockfile and grab the date/errfile pairs.
141:     unset($lockfile_data);
142:     while (!feof($lockfile)) {
143:         $buffer = fgets($lockfile);
144:         if ($buffer == '') continue;  // EOF line is empty
145:         $match = preg_match('#^([0-9]{4}-[0-9]{2}-[0-9]{2}'.
146:                 'T[0-9]{2}:[0-9]{2}:[0-9]{2}(?:\+|-)[0-9]{4})'.
147:                 '\s+(.+)$#', $buffer, $matches);
148:         if (!$match) {
149:             error_log('FAIL: preg_match could not match '.
150:                     'expected pattern in error log file, in '.
151:                     __FILE__ .'');
152:             continue;
153:         }
154:         $lockfile_data[$matches[2]] = $matches[1]; 
155:     }
156:     
157:     // If an alert date exists for the current $errfile value,
158:     // then calculate whether or not the last error alert was
159:     // sent (by email) less than twenty-four hours ago. If so,
160:     // we don't need to send another alert right now.
161:     $need_to_email_alert = true; // starting assumption
162:     if (count($lockfile_data) > 0) {
163:         if (isset($lockfile_data[$errfile])) {
164:             $alert_date = date_create($lockfile_data[$errfile]);

171:                 die();

173:             $alert_timestamp = date_format($alert_date, 'U');

186:                 $need_to_email_alert = false;

188:         }
189:     }

191:     
192:     /*
193:      * Stage three: send email and update lockfile, if necessary.
194:      */

196:     // Either we've never sent an alert about this page before,
197:     // or the last alert was sent more than 24 hours ago.

202:         // Write out an updated version of the locking file after
203:         // updating the alert date for this $errfile entry.
204:         $lockfile_data[$errfile] = gmdate(DATE_ISO8601);
205:         ftruncate($lockfile, 0);  // we want to start from blank

228:                     "\t".$errfile.' (line '.$errline.')'.$rtnl.

235:                     'by this page in the next 24 hours. '.

239:                     'live error log.)';

158:    * json_decode *nearly* works in some cases, but this tends to be more reliable

162:   public function decodeArray($text) {
163:     $limit = strlen($text) - 1;
164:     $output = array();
165:     $offset = 1;
166:     
167:     if ('{}' != $text)
168:       do {
169:         if ('{' != $text{$offset}) {
170:           preg_match("/(\\{?\"([^\"\\\\]|\\\\.)*\"|[^,{}]+)+([,}]+)/", $text, $match, 0, $offset);
171:           $offset += strlen($match[0]);
172:           $output[] = ( '"' != $match[1]{0} ? $match[1] : stripcslashes(substr($match[1], 1, -1)) );
173:           if ('},' == $match[3])
174:             return $offset;
175:         }
176:         else
177:           $offset = pg_array_parse($text, $output[], $limit, $offset + 1);
178:       }
179:       while ($limit > $offset);

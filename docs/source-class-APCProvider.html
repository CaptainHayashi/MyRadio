  7: /**
  8:  * APCProvider provides in-memory caching for PHP resources to increase page load times
  9:  * 
 10:  * APCProvider was the first CacheProvider implementation in MyURY. It enables Models to send cache commands to it
 11:  * which are then stored using the APC plugin automatically. In order for this class to work correctly, the server
 12:  * needs the APC PHP plugin installed on the server. It will throw an Exception if it is not.
 13:  *
 14:  * @author Lloyd Wallis <lpw@ury.org.uk>
 15:  * @version 21072012
 16:  * @package MyURY_Core
 17:  */
 18: class APCProvider implements CacheProvider {
 19:   /**
 20:    * A variable to store the singleton instance
 21:    * @var APCProvider storage for the only APCProvider instance
 22:    */
 23:   private static $me;
 24:   
 25:   /**
 26:    * Stores whether caching should be used. If not, it does not do anything on function calls
 27:    * @var boolean
 28:    */
 29:   private $enable;
 30:   
 31:   /**
 32:    * Constructs the Unique instance of the CacheProvider for use. Private so that instances cannot be used in ways
 33:    * other than those intended
 34:    * @param boolean $enable Whether caching is actually enabled in this request. Default true
 35:    * @throws MyURYException Will throw a MyURYException if the APC extension is not loaded
 36:    */
 37:   private function __construct($enable = true) {
 38:     $this->enable = $enable;
 39:     if ($enable && !function_exists('apc_store')) {
 40:       //Functions not available. If this is caught upstream, just disable
 41:       throw new MyURYException('Cache is enabled but selected CacheProvider does not have required prerequisites (Is APC Extension installed and loaded?)');
 42:       $this->enable = false;
 43:     }
 44:   }
 45:   
 46:   /**
 47:    * Stores an object in the APC User Object Cache
 48:    * 
 49:    * @param String $key The unique name of the object to store. Ideally, this would use myury_{module}_{name}
 50:    * @param mixed $value The data to store
 51:    * @param int $expires The number of seconds this cache entry is valid for. Default is to last forever (0)
 52:    * @return boolean Whether the operation was successful (returns false if caching disabled)
 53:    * @assert ('myury_core_test', 'test value', 0) == true
 54:    */
 55:   public function set($key, $value, $expires = 0) {
 56:     if (!$this->enable) return false;
 57:     return apc_store($key, $value, $expires);
 58:   }
 59:   
 60:   /**
 61:    * Reads a previously stored value from the APC User Object Cache and returns it
 62:    * 
 63:    * @param String $key The unique name of the object to fetch
 64:    * @return mixed The value of the store, or false on failure
 65:    * @assert ('myury_core_test') == 'test value'
 66:    */
 67:   public function get($key) {
 68:     if (!$this->enable) return false;
 69:     return apc_fetch($key);
 70:   }
 71:   
 72:   /**
 73:    * Deletes a previously stored value from the APC User Object Cache
 74:    * 
 75:    * @param String $key The unique name of the object to delete
 76:    * @return boolean Returns whether the operaion was a success
 77:    * @assert ('myury_core_test') == true
 78:    */
 79:   public function delete($key) {
 80:     if (!$this->enable) return false;
 81:     return apc_delete($key);
 82:   }
 83:   
 84:   /**
 85:    * This function currently does nothing as it does not have an obvious APC implementation
 86:    * 
 87:    * The theory is that this will completely wipe the APC User Object Cache
 88:    * 
 89:    * @todo find a way to implement this in APC
 90:    */
 91:   public function purge() {
 92:     if (!$this->enable) return false;
 93:     return true;
 94:   }
 95:   
 96:   /**
 97:    * Returns the Singleton instance of this class, creating it if necessary
 98:    * 
 99:    * @return APCProvider
100:    */
101:   public static function getInstance() {
102:     if (!self::$me) {
103:       self::$me = new self(Config::$cache_enable);
104:     }
105:     return self::$me;
106:   }
107:   
108:   /**
109:    * Prevent copies being unintentionally made
110:    * @throws MyURYException
111:    */
112:   public function __clone() {
113:     throw new MyURYException('Attempted to clone a singleton');
114:   }
115: }

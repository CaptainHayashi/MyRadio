107:           if ($section['columnid'] != $column['columnid'])

208:   /**
209:    * Detects module/action/service links and rewrites
210:    * This is a method so it can easily be changed if Apache rewrites
211:    * @param String $url The URL to parse
212:    * @todo Rewrite this to make sense
213:    */
214:   private function parseURL($url, $return = 'url') {
215:     $exp = explode(',', $url);
216:     if (preg_match('/^(module|service)=/', $exp[0]) == 1) {
217:       //It can be rewritten!
218:       $url = '?' . $exp[0];
219:       if (isset($exp[1])) {
220:         //An action is defined!
221:         $url .= '&' . $exp[1];
222:         if (isset($exp[2])) {
223:           //An additional query string
224:           //This could be multiple variables separated by &
225:           $url .= '&' . $exp[2];
226:         }
227:       }
228:     } else {
229:       //It's not a rewritable
230:       if ($return !== 'url')
231:         return null;
232:     }
233:     if ($return === 'service') {
234:       if (preg_match('/^service=/', $exp[0]) == 1) {
235:         return str_replace('service=', '', $exp[0]);
236:       } else {
237:         return 'MyURY';
238:       }
239:     } elseif ($return === 'module') {
240:       if (preg_match('/^module=/', $exp[0]) == 1) {
241:         return str_replace('module=', '', $exp[0]);
242:       } else {
243:         return 'Core';
244:       }
245:     } elseif ($return === 'action') {
246:       if (isset($exp[1])) {
247:         return str_replace('action=', '', $exp[1]);
248:       } else {
249:         return 'default';
250:       }
251:     }
252:     return $url;
253:   }
